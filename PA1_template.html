Imputing missing values

## [1] 2304

na.indices <- which(is.na(activity$steps))
activity.imputed <- activity
for (ix in na.indices) {
    activity.imputed[ix, ]$steps <- as.list(steps.by.interval)[[as.character(activity[ix, 
        ]$interval)]]
}
